110// Q: Why is there a __nop() before __debugbreak()?
111// A: VS' debug engine has a bug where it will silently swallow explicit
112// breakpoint interrupts when single-step debugging either line-by-line or
113// over call instructions. This can hide legitimate reasons to trap. Asserts
114// for example, which can appear as if the did not fire, leaving a programmer
115// unknowingly debugging an undefined process.
116#define PLATFORM_BREAK() (__nop(), __debugbreak())
